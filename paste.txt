// Delete markers for stale tracks so RViz doesn't keep them
for (int id : to_del) {
  visualization_msgs::Marker del;
  del.header.frame_id = msg->header.frame_id;
  del.header.stamp = now;
  del.ns = "tracked_boxes";
  del.id = id;
  del.action = visualization_msgs::Marker::DELETE;
  box_arr.markers.push_back(del);

  visualization_msgs::Marker del2 = del;
  del2.ns = "track_labels";
  label_arr.markers.push_back(del2);
}
