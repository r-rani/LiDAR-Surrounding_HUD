cmake_minimum_required(VERSION 3.0.2)
project(gr14)

add_compile_options(-std=c++11)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  sensor_msgs
  pcl_ros
  pcl_conversions
  visualization_msgs
)

catkin_package(
  CATKIN_DEPENDS roscpp sensor_msgs pcl_ros pcl_conversions visualization_msgs
)

include_directories(
  ${catkin_INCLUDE_DIRS}
)

add_executable(unitree_outlines src/outlines.cpp)

target_link_libraries(unitree_outlines
  ${catkin_LIBRARIES}
)
