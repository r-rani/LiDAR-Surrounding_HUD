cmake_minimum_required(VERSION 3.0.2)
project(gr14)

add_compile_options(-O3 -std=c++14)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  sensor_msgs
  visualization_msgs
  pcl_ros
  pcl_conversions
)

catkin_package()

include_directories(
  ${catkin_INCLUDE_DIRS}
)

add_executable(object_outlines src/object_outlines.cpp)
target_link_libraries(object_outlines ${catkin_LIBRARIES})
