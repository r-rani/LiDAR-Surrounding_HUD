cmake_minimum_required(VERSION 3.0.2)
project(gr14)

add_compile_options(-O3 -std=c++14)

find_package(catkin REQUIRED COMPONENTS
  roscpp
  sensor_msgs
  visualization_msgs
  pcl_ros
  pcl_conversions
)

find_package(PCL REQUIRED COMPONENTS
  common
  io
  filters
  segmentation
  kdtree
  search
  sample_consensus
)

catkin_package()

include_directories(
  ${catkin_INCLUDE_DIRS}
  ${PCL_INCLUDE_DIRS}
)

add_definitions(${PCL_DEFINITIONS})

add_executable(object_boxes_tracker src/object_boxes_tracker.cpp)
target_link_libraries(object_boxes_tracker
  ${catkin_LIBRARIES}
  ${PCL_LIBRARIES}
)
